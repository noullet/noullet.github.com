Classes made easy


	import groovy.transform.*

	@EqualsAndHashCode
	@ToString
	@TupleConstructor
	class Cat {
	    def nickname
	    def isPlottingToKillYou
	}

	def felix = new Cat("Felix", false)

	println felix // output: Cat(Felix, false)
	assert felix.nickname == "Felix"
	assert felix.isPlottingToKillYou == false
